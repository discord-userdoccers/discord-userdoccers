# DSL

```ts pmo
/**
 * Emoji object **some**
 * _markdown_
 */
interface Emoji {
  /**
   * The [ID of the emoji](/reference#cdn-formatting)
   * hii
   *
   * @note hello
   * @deprecated no more ids, we only use names now
   */
  id: Snowflake;
  /**
   * The name of the emoji (2-32 characters)
   *
   * @note This may be `null` in case of a deleted _custom_ emoji.
   * @note hiii
   * @column Changes: **SomeObject**
   * @deleted
   */
  name: string | Integer | null;
  foo: [string, boolean];
  /**
   * The roles allowed to use the emoji
   *
   * @deprecated
   * @column Foo: aaa
   */
  roles?: Snowflake[];
  /**
   * @note *1
   */
  user?: Resources.User.User;
  assets: Record<Snowflake, Resources.Application.ApplicationAsset[]>;
}

enum VisibilityType {
  /**
   * @column Foo: SomeObject
   */
  NONE = 0,
  EVERYONE = 1,
}

enum PlatformType {
  Web = "web",
  Mobile = "mobile",
}

enum UserFlags {
  ADMIN = 1 << 0,
  STUPID = 1 << 2,
}
```
