# DSL

```ts pmo
/**
 * Emoji object
 */
interface Emoji {
  /**
   * The [ID of the emoji](/referenceeee#cdn-formatting)
   *
   * @deprecated no more ids, we only use names now
   */
  id: Snowflake;
  /**
   * The name of the emoji (2-32 characters)
   *
   * @note This may be `null` in case of a deleted custom emoji.
   * @deleted
   */
  name: string | null;
  /**
   * The roles allowed to use the emoji
   *
   * @deprecated
   */
  roles?: Snowflake[];
  user?: Resources.User.User;
  assets: Record<Snowflake, Resources.Application.ApplicationAsset[]]
}

enum Visibility {
  NONE = 0,
  EVERYONE = 1
}

enum Platform {
  Web = "web",
  Mobile = "mobile"
}

enum UserFlags {
  ADMIN = 1 << 0,
  STUPID = 1 << 2
}
```
